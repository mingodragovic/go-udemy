func factorial(n int) int {
	// base case
	if n == 0 {
		return 1
	}
	return n * factorial(n-1)
}
func sumOfDigits(n int) int {
	if n < 10 {
		return n
	}
	return n % 10 * sumOfDigits(n/10)
}
